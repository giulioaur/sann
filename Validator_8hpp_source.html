36  // Those classes are a base class for Estimators that exploit the early stop criteria. They are abstract
37  // class, so they need to be extended by another class to be instantiated. The aim of this behaviour is to
38  // allow the user to use different kind of error on different model selection. The error is the error computed
39  // during the training, not the loss of the Validator, that is specified on creation.
40  // They will be passed to the training method.
41  // NB: The model selection is better done if the error plotted by those estimator and the one minimized by
42  // the networks is the same.
43 
44  // To extends this class the user have to implement three methods: update(), clone() and finalize().
45  // The first one is described in the Estimator header file, the second one returns a copy of the Estimator
46  // and is needed to hide the copy constructor, the last one is called right before starting the plot method
47  // (that could not be overrided).
48  // The attribute errorTreshold could be changed to edit the threshold at which the training error has to
49  // stop.
50  // Due to speed reason, the plot is done by writing on a file a string with all the csv data.
51  class TrValidEstimator : public Estimator{
52  private:
53  bool earlyStop = false;
54  std::size_t epoch, oldEpoch;
55  std::string filename, results = "", oldResults;
56  double oldError, oldAccuracy;
57  protected:
58  double accuracy, error = 1, errorThreshold = 10e-7;
59  public:
60  TrValidEstimator(const std::string &filename) : filename(filename) {}
61  virtual std::unique_ptr<TrValidEstimator> clone(const std::string &filename) const = 0;
62  virtual void init(const std::size_t epoch){
63  this->accuracy = this->error = 0; this->epoch = epoch;
64  this->results += std::to_string(epoch) + ',';
65  }
66  bool stoppingCriteria(){ return this->error <= errorThreshold || earlyStop; }
67  virtual void finalize() = 0;
68  void plot(){
69  this->finalize();
70  this->results += std::to_string(error) + ',' + std::to_string(accuracy) + '\n';
71  }
72  void terminate(){
73  if(this->earlyStop){
74  utility::FileManager::writeFile(FILES_DIR + "validation/" + filename + ".csv", this->oldResults);
75  this->epoch = this->oldEpoch;

78  utility::FileManager::writeFile(FILES_DIR + "validation/" + filename + ".csv", this->results);
79  }
80  virtual double getAccuracy() const{ return this->earlyStop ? this->oldAccuracy : this->accuracy; }
81  virtual double getError() const{ return this->earlyStop ? this->oldError : this->error; }
82  virtual double getEpoch() const{ return this->earlyStop ? this->oldEpoch : this->epoch; }
83  void setEarlyStop(){ this->earlyStop = true; }
84  void saveResults(){
85  this->oldResults = this->results; this->oldEpoch = this->epoch;
86  this->oldError = this->error; this->oldAccuracy = this->accuracy;
87  }

90  // To implement the early stop criteria, another estimator is introduced for the validation set.
91  // This estimator keep tracks of the error of the validation set, if this error does not decrease after a
92  // certain number of epochs, the estimator stop the training. To do that it sets the attribute earlyStop
93  // of the training estimator to true.
94  // The three methods update(), clone() and finalize() have to be implemented (read above).
95  // Three attribute could be changed: earlyStep changes the number of step in which earlyStop criteria is
96  // checked, earlyThreshold changes the number of bad checking has to be done before set earlyStop to true,
97  // errorThreshold is the difference between last saved error and the current one to be sure that we are
98  // going into overfitting.
99  class VdValidEstimator : public Estimator{
100  private:
101  TrValidEstimator &trEst;
102  std::size_t currIteration = 0, epoch;
103  double oldError = MAX_DOUBLE, oldAccuracy = MAX_DOUBLE;
104  protected:
105  std::size_t earlyStep = 2, earlyThreshold = 1000;
106  double error, accuracy, errorThreshold = 0.2;
107  public:
108  VdValidEstimator(TrValidEstimator &trEst) : trEst(trEst){}
109  virtual std::unique_ptr<VdValidEstimator> clone(TrValidEstimator &trEst) const = 0;
110  virtual void init(const std::size_t epoch){
111  this->accuracy = this->error = 0; this->epoch = epoch;
112  }
113  bool stoppingCriteria(){ return false; }
114  virtual void finalize() = 0;
115  void plot(){
116  this->finalize();
117  // Each n epoch check if the mse is decreasing. If so, init current iteration to 0.
118  if(this->epoch % this->earlyStep == 0){
119  if(this->error < this->oldError){
120  this->oldError = this->error; this->oldAccuracy = this->accuracy;
121  this->currIteration = 0;
122  this->trEst.saveResults();
123  }
124  else if(this->error - this->oldError > errorThreshold){
125  this->currIteration++;
126  }
127  }
128 
129  // If the number of iteration since mse decresed is more than the threshold, set the
130  // earlystop attribute of the training estimator to true.
131  if(this->currIteration >= this->earlyThreshold)
132  this->trEst.setEarlyStop();
133  }
134  virtual void terminate(){ }
135  size_t getEpoch(){ return this->epoch; }
136  };
